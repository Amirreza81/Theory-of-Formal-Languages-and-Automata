\textbf{پرسش پنجم}\\[0.1in]
نشان دهید زبان‌های زیر منظم هستند.
\begin{enumerate}
    \item[1.] $L = \{ abwba : w \in \{ a, b \}^*\}$.
    \item [2.]
    $L = \{w \in \{ a, b \}^* : aba \leq_{sub} w \wedge 
    bba \nleq_{sub} w \} $.
    \item[3.]
    $L = \{w \in \{ a, b \}^* : [n_a(w)\;\; mod\;\; 3] > [n_b(w) \;\;mod\;\; 2]\}$.
\end{enumerate}
